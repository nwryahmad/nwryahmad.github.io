// C++ code // #include LiquidCrystal lcd = LiquidCrystal(2,3,4,5,6,7);
int red = 11;
int yelo = 12;
int green = 13;
int buzzor = 8;
int gas = A5;
void setup()
{
pinMode(red,OUTPUT);
pinMode(yelo,OUTPUT);
pinMode(green,OUTPUT);
pinMode(buzzor,OUTPUT);
pinMode(gas,INPUT);
lcd.begin(16,2);
Serial.begin(9600);

} void loop()
{
int value = analogRead(gas);
Serial.println(value);
if (value > 90 && value < 150){
digitalWrite(green,HIGH);
lcd.setCursor(0,0);
lcd.print("green is ON");
lcd.setCursor(0,1);
lcd.print(value);
noTone(buzzor);
}else if(value > 200 && value < 300){
digitalWrite(green,LOW);
digitalWrite(yelo,HIGH);
lcd.clear();
lcd.setCursor(0,0);
lcd.print("Yelo is ON");
lcd.setCursor(0,1);
lcd.print(value);
tone(buzzor,50);
digitalWrite(buzzor,HIGH);
} }
