#include
Servo servo;
int flex = A0;
int motionState = 0;
void setup()
{
pinMode(flex, INPUT);
pinMode(13, OUTPUT);// led y taybat ba motion sensor aka
pinMode(2, INPUT);// taybat ba motion sensor
servo.attach(A1);
Serial.begin(9600);
}
void loop()
{
int flexVal = analogRead(flex);
motionState = digitalRead(2);
if (motionState == HIGH) {
digitalWrite(13, HIGH);
Serial.println("Sensor activated!");
} else {
digitalWrite(13, LOW);
}
delay(10);
flexVal = map(flexVal,52,256,0,180);
servo.write(flexVal);
}
