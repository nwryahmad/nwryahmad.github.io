import 'package:flutter/material.dart';
void main() => runApp(MaterialApp(
home: hello(),
));
class hello extends StatefulWidget {
const hello({super.key});
@override
State createState() => _helloState();
}
class _helloState extends State {
int counter = 0;
@override
Widget build(BuildContext context) {
return Scaffold(
body: Container(
height: 400,
child: Row(
mainAxisAlignment: MainAxisAlignment.center,
children: [
ElevatedButton(
onPressed: () {
setState(() {
counter++;
});
},
child: Icon(
Icons.add,
size: 20,
),
style: ElevatedButton.styleFrom(
shape: CircleBorder(),
),
),
Text(
'$counter',
style: TextStyle(fontSize: 40),
),
ElevatedButton(
onPressed: () {
setState(() {
counter--; });
},
child: Icon(
Icons.remove,
size: 20,
),
style: ElevatedButton.styleFrom(
shape: CircleBorder(),
),
),
],
),
),
);
}
}
